pages/Workflows.js
import React, { useState } from "react";
import { base44 } from "@/api/base44Client";
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Textarea } from "@/components/ui/textarea";
import { Switch } from "@/components/ui/switch";
import { Badge } from "@/components/ui/badge";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/dialog";
import {
  Play,
  Pause,
  Plus,
  Trash2,
  Edit,
  Copy,
  Activity,
  Mail,
  MessageSquare,
  CheckSquare,
  Clock,
  Zap,
  Settings,
  TrendingUp,
  Eye,
  Save,
  Workflow,
  Search,
  Folder,
  Sparkles,
  Bot
} from "lucide-react";
import { format } from "date-fns";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";
import WorkflowActivityLog from "@/components/workflows/WorkflowActivityLog";

export default function Workflows() {
  const [user, setUser] = React.useState(null);
  const [myCompany, setMyCompany] = React.useState(null);

  React.useEffect(() => {
    base44.auth.me().then(setUser).catch(() => {});
  }, []);

  const { data: companies = [] } = useQuery({
    queryKey: ['companies'],
    queryFn: () => base44.entities.Company.list("-created_date"),
    initialData: [],
  });

  const { data: staffProfiles = [] } = useQuery({
    queryKey: ['staff-profiles'],
    queryFn: () => user ? base44.entities.StaffProfile.filter({ user_email: user.email }) : [],
    enabled: !!user,
    initialData: [],
  });

  React.useEffect(() => {
    if (user && companies.length > 0) {
      const ownedCompany = companies.find(c => c.created_by === user.email);
      if (ownedCompany) {
        setMyCompany(ownedCompany);
        return;
      }
      const staffProfile = staffProfiles[0];
      if (staffProfile?.company_id) {
        setMyCompany(companies.find(c => c.id === staffProfile.company_id));
      }
    }
  }, [user, companies, staffProfiles]);

  const [showDialog, setShowDialog] = useState(false);
  const [editingWorkflow, setEditingWorkflow] = useState(null);
  const [showActionsEditor, setShowActionsEditor] = useState(false);
  const [showAIPrompt, setShowAIPrompt] = useState(false);
  const [editingAIWorkflow, setEditingAIWorkflow] = useState(null);
  const [aiPrompt, setAIPrompt] = useState("");
  const [isGenerating, setIsGenerating] = useState(false);
  const [formData, setFormData] = useState({
    workflow_name: "",
    description: "",
    trigger_type: "estimate_created",
    is_active: true,
    actions: []
  });

  const queryClient = useQueryClient();

  const { data: workflows = [] } = useQuery({
    queryKey: ['workflows', myCompany?.id],
    queryFn: () => myCompany ? base44.entities.Workflow.filter({ 
      company_id: { $in: [myCompany.id, '695944e3c1fb00b7ab716c6f'] } 
    }, "-created_date") : [],
    enabled: !!myCompany,
    initialData: [],
  });

  const { data: executions = [] } = useQuery({
    queryKey: ['workflow-executions'],
    queryFn: () => base44.entities.WorkflowExecution.filter({ status: 'active' }),
    initialData: [],
  });

  const [testResults, setTestResults] = useState({});
  const [searchQuery, setSearchQuery] = useState("");
  const [folderFilter, setFolderFilter] = useState("all");

  const folders = React.useMemo(() => {
    const set = new Set((workflows || []).map(w => (w.folder || '').trim()).filter(Boolean));
    return Array.from(set);
  }, [workflows]);

  const filteredWorkflows = React.useMemo(() => {
    const q = (searchQuery || '').toLowerCase().trim();
    return (workflows || []).filter(w => {
      const inFolder = folderFilter === 'all' || w.folder === folderFilter;
      const haystack = `${w.workflow_name || ''} ${w.description || ''} ${w.trigger_type || ''} ${(w.tags || []).join(' ')}`.toLowerCase();
      const matches = q === '' || haystack.includes(q);
      return inFolder && matches;
    });
  }, [workflows, searchQuery, folderFilter]);

  const createMutation = useMutation({
    mutationFn: (data) => base44.entities.Workflow.create(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['workflows'] });
      setShowDialog(false);
      resetForm();
    },
  });

  const updateMutation = useMutation({
    mutationFn: ({ id, data }) => base44.entities.Workflow.update(id, data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['workflows'] });
    },
  });

  const deleteMutation = useMutation({
    mutationFn: (id) => base44.entities.Workflow.delete(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['workflows'] });
    },
  });

  const testMutation = useMutation({
    mutationFn: async (workflowId) => {
      const response = await base44.functions.invoke('testWorkflow', { workflowId });
      return response.data;
    },
  });

  const resetForm = () => {
    setFormData({
      workflow_name: "",
      description: "",
      trigger_type: "estimate_created",
      is_active: true,
      actions: []
    });
    setEditingWorkflow(null);
    setShowActionsEditor(false);
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    if (editingWorkflow) {
      updateMutation.mutate({ id: editingWorkflow.id, data: formData });
      setShowDialog(false);
      setShowActionsEditor(false);
      resetForm();
    } else {
      createMutation.mutate(formData);
    }
  };

  const handleEdit = (workflow) => {
    setEditingWorkflow(workflow);
    setFormData(workflow);
    setShowDialog(true);
    setShowActionsEditor(true);
  };

  const handleToggle = (workflow) => {
    updateMutation.mutate({
      id: workflow.id,
      data: { ...workflow, is_active: !workflow.is_active }
    });
  };

  const handleDelete = (id) => {
    if (window.confirm('Are you sure you want to delete this workflow?')) {
      deleteMutation.mutate(id);
    }
  };

  const handleTest = async (workflow) => {
    setTestResults(prev => ({
      ...prev,
      [workflow.id]: { status: 'loading', message: 'Running test...' }
    }));

    try {
      const response = await base44.functions.invoke('testWorkflow', {
        workflowId: workflow.id
      });
      
      const data = response.data || {};
      const emailsSent = data.emailsSent || 0;
      const smsSent = data.smsSent || 0;
      const notifs = data.notificationsCreated || 0;
      const errors = data.errors || [];
      
      let resultMessage = `✅ Test Complete!\n`;
      resultMessage += `📧 Emails sent: ${emailsSent}\n`;
      resultMessage += `📱 SMS sent: ${smsSent}\n`;
      resultMessage += `🔔 Notifications: ${notifs}`;
      
      if (errors.length > 0) {
        resultMessage += `\n\n⚠️ Errors:\n` + errors.map(e => `• ${e.type}: ${e.message}`).join('\n');
      }
      
      setTestResults(prev => ({
        ...prev,
        [workflow.id]: { 
          status: errors.length > 0 ? 'warning' : 'success', 
          message: resultMessage,
          timestamp: new Date().toLocaleTimeString(),
          details: data
        }
      }));
      
      queryClient.invalidateQueries({ queryKey: ['notifications'] });
      
    } catch (error) {
      setTestResults(prev => ({
        ...prev,
        [workflow.id]: { 
          status: 'error', 
          message: `❌ Test failed: ${error.message}`,
          timestamp: new Date().toLocaleTimeString()
        }
      }));
    }
  };

  const getTriggerIcon = (trigger) => {
    if (trigger.includes('estimate') || trigger.includes('proposal')) return <Mail className="w-4 h-4" />;
    if (trigger.includes('lead')) return <TrendingUp className="w-4 h-4" />;
    if (trigger.includes('invoice') || trigger.includes('payment')) return <CheckSquare className="w-4 h-4" />;
    if (trigger.includes('task') || trigger.includes('project')) return <CheckSquare className="w-4 h-4" />;
    if (trigger.includes('appointment') || trigger.includes('event') || trigger.includes('inspection')) return <Clock className="w-4 h-4" />;
    if (trigger.includes('customer') || trigger.includes('staff')) return <Activity className="w-4 h-4" />;
    if (trigger === 'ai_communication_logged') return <MessageSquare className="w-4 h-4" />;
    return <Zap className="w-4 h-4" />;
  };

  const updateAction = (index, field, value) => {
    const newActions = [...(formData.actions || [])];
    newActions[index] = { ...newActions[index], [field]: value };
    setFormData({ ...formData, actions: newActions });
  };

  const addAction = () => {
    const newAction = {
      step: (formData.actions?.length || 0) + 1,
      action_type: 'send_email',
      delay_minutes: 0,
      recipient: 'yicnteam@gmail.com'
    };
    setFormData({ ...formData, actions: [...(formData.actions || []), newAction] });
  };

  const removeAction = (index) => {
    const newActions = formData.actions.filter((_, i) => i !== index);
    setFormData({ ...formData, actions: newActions });
  };

  const activeWorkflows = workflows.filter(w => w.is_active).length;
  const totalExecutions = workflows.reduce((sum, w) => sum + (w.execution_count || 0), 0);
  const runningNow = executions.length;

  return (
    <div className="p-6 space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-bold text-gray-900">Workflow Automation</h1>
          <p className="text-gray-500 mt-1">Automate your sales process and follow-ups</p>
        </div>

        <div className="flex gap-2">
          <Button
            variant="outline"
            onClick={async () => {
              if (!myCompany?.id) { alert('No company found'); return; }
              if (window.confirm('Setup default workflows for all key events?')) {
                try {
                  const response = await base44.functions.invoke('setupDefaultWorkflows', { companyId: myCompany.id });
                  if (response.data.success) {
                    alert('✅ ' + response.data.message);
                    queryClient.invalidateQueries({ queryKey: ['workflows'] });
                  }
                } catch (error) { alert('❌ Error: ' + error.message); }
              }
            }}
            className="bg-purple-600 hover:bg-purple-700 text-white"
          >
            <Zap className="w-4 h-4 mr-2" />
            Setup Defaults
          </Button>

          <Button 
            className="bg-gradient-to-r from-indigo-500 to-pink-500 hover:from-indigo-600 hover:to-pink-600 text-white border-0"
            onClick={() => setShowAIPrompt(true)}
          >
            <Sparkles className="w-4 h-4 mr-2" />
            Create with AI
          </Button>

          <Dialog open={showDialog && !editingWorkflow} onOpenChange={(open) => {
            setShowDialog(open);
            if (!open) resetForm();
          }}>
            <DialogTrigger asChild>
              <Button className="bg-blue-600 hover:bg-blue-700" onClick={() => setEditingWorkflow(null)}>
                <Plus className="w-4 h-4 mr-2" />
                Create Workflow
              </Button>
            </DialogTrigger>
          </Dialog>
        </div>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-4 gap-4">
        <Card>
          <CardContent className="p-6">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm text-gray-500">Total Workflows</p>
                <p className="text-3xl font-bold text-gray-900">{workflows.length}</p>
              </div>
              <Settings className="w-8 h-8 text-gray-400" />
            </div>
          </CardContent>
        </Card>
        <Card>
          <CardContent className="p-6">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm text-gray-500">Active Workflows</p>
                <p className="text-3xl font-bold text-green-600">{activeWorkflows}</p>
              </div>
              <Activity className="w-8 h-8 text-green-400" />
            </div>
          </CardContent>
        </Card>
        <Card>
          <CardContent className="p-6">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm text-gray-500">Total Executions</p>
                <p className="text-3xl font-bold text-blue-600">{totalExecutions}</p>
              </div>
              <Zap className="w-8 h-8 text-blue-400" />
            </div>
          </CardContent>
        </Card>
        <Card>
          <CardContent className="p-6">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm text-gray-500">Running Now</p>
                <p className="text-3xl font-bold text-purple-600">{runningNow}</p>
              </div>
              <Clock className="w-8 h-8 text-purple-400" />
            </div>
          </CardContent>
        </Card>
      </div>

      {showDialog && editingWorkflow && !showActionsEditor && (
        <Dialog open={showDialog && editingWorkflow && !showActionsEditor} onOpenChange={(open) => {
          if (!open) { setShowDialog(false); setEditingWorkflow(null); resetForm(); }
        }}>
          <DialogContent className="max-w-4xl max-h-[90vh] overflow-y-auto">
            <DialogHeader>
              <DialogTitle className="flex items-center gap-2">
                <Workflow className="w-5 h-5 text-blue-600" />
                {formData.workflow_name}
              </DialogTitle>
            </DialogHeader>
            <div className="space-y-6">
              <div className="bg-gray-50 p-4 rounded-lg space-y-2">
                <div className="flex items-center gap-2">
                  <span className="text-sm font-semibold text-gray-700">Trigger:</span>
                  <Badge variant="outline">{formData.trigger_type?.replace(/_/g, ' ')}</Badge>
                </div>
                {formData.description && <p className="text-sm text-gray-600">{formData.description}</p>}
                <div className="flex items-center gap-2">
                  <span className="text-sm font-semibold text-gray-700">Status:</span>
                  <Badge className={formData.is_active ? "bg-green-100 text-green-700" : "bg-gray-100 text-gray-700"}>
                    {formData.is_active ? 'Active' : 'Inactive'}
                  </Badge>
                </div>
              </div>

              {((formData.actions && formData.actions.length > 0) || (formData.steps && formData.steps.length > 0)) ? (
                <div>
                  {(() => {
                    const displayActions = formData.actions || formData.steps || [];
                    return (
                      <>
                        <h3 className="text-lg font-semibold mb-4 flex items-center gap-2">
                          <Clock className="w-5 h-5 text-blue-600" />
                          Automation Timeline
                        </h3>
                        <div className="space-y-6">
                          {displayActions.map((action, idx) => {
                            let cumulativeDelay = 0;
                            for (let i = 0; i <= idx; i++) {
                              cumulativeDelay += (displayActions[i].delay_minutes || displayActions[i].config?.delay_minutes || 0);
                            }
                            const hours = Math.floor(cumulativeDelay / 60);
                            const days = Math.floor(hours / 24);
                            const remainingHours = hours % 24;
                            let timeLabel = '';
                            if (cumulativeDelay === 0) timeLabel = '🟢 Immediately';
                            else if (days > 0) timeLabel = `⏰ After ${days} day${days > 1 ? 's' : ''} ${remainingHours > 0 ? `${remainingHours}h` : ''}`;
                            else if (hours > 0) timeLabel = `⏰ After ${hours} hour${hours > 1 ? 's' : ''}`;
                            else timeLabel = `⏰ After ${cumulativeDelay} minutes`;

                            return (
                              <div key={idx} className="relative z-10 bg-white border-2 border-blue-300 rounded-xl p-5 shadow-lg">
                                <div className="absolute -top-3 left-4 bg-gradient-to-r from-blue-600 to-purple-600 text-white px-4 py-1 rounded-full text-sm font-bold shadow-md">
                                  {timeLabel}
                                </div>
                                <div className="flex items-start gap-4 mt-2">
                                  <div className="flex-shrink-0 w-14 h-14 bg-gradient-to-br from-blue-600 to-purple-600 text-white rounded-full flex items-center justify-center font-bold text-xl shadow-lg border-4 border-white">
                                    {idx + 1}
                                  </div>
                                  <div className="flex-1">
                                    <div className="flex items-center gap-3 mb-3">
                                      {((action.action_type || action.type) === 'send_email') && <Mail className="w-6 h-6 text-blue-600" />}
                                      {((action.action_type || action.type) === 'send_sms') && <MessageSquare className="w-6 h-6 text-green-600" />}
                                      {((action.action_type || action.type) === 'create_task') && <CheckSquare className="w-6 h-6 text-purple-600" />}
                                      <span className="font-bold text-gray-900 text-xl capitalize">
                                        {(action.action_type || action.type)?.replace(/_/g, ' ')}
                                      </span>
                                    </div>
                                    {(action.recipient || action.config?.recipient) && (
                                      <div className="mb-3 bg-blue-50 border border-blue-200 rounded-lg p-3">
                                        <span className="text-sm font-bold text-gray-700">📧 Send To: </span>
                                        <span className="text-sm text-blue-700 font-semibold">{action.recipient || action.config?.recipient}</span>
                                      </div>
                                    )}
                                    {(action.email_subject || action.config?.email_subject) && (
                                      <div className="mb-3">
                                        <p className="text-sm font-bold text-gray-700 mb-2">📌 Email Subject:</p>
                                        <p className="text-base text-gray-900 bg-blue-50 p-3 rounded-lg border-l-4 border-blue-500 font-medium">
                                          {action.email_subject || action.config?.email_subject}
                                        </p>
                                      </div>
                                    )}
                                    {(action.sms_message || action.config?.sms_message) && (
                                      <div className="mb-3">
                                        <p className="text-sm font-bold text-gray-700 mb-2">💬 SMS Message:</p>
                                        <div className="text-sm text-gray-800 bg-green-50 p-4 rounded-lg border-l-4 border-green-500 whitespace-pre-wrap">
                                          {action.sms_message || action.config?.sms_message}
                                        </div>
                                      </div>
                                    )}
                                  </div>
                                </div>
                              </div>
                            );
                          })}
                        </div>
                      </>
                    );
                  })()}
                </div>
              ) : (
                <div className="text-center py-8 bg-gray-50 rounded-lg">
                  <Activity className="w-12 h-12 mx-auto mb-3 text-gray-300" />
                  <p className="text-gray-500">No automation steps configured yet</p>
                </div>
              )}

              <div className="flex justify-end gap-3 pt-4 border-t">
                <Button type="button" variant="outline" onClick={() => { setShowDialog(false); setEditingWorkflow(null); resetForm(); }}>
                  Close
                </Button>
                <Button type="button" onClick={() => setShowActionsEditor(true)} className="bg-blue-600 hover:bg-blue-700">
                  <Edit className="w-4 h-4 mr-2" />
                  Edit Workflow
                </Button>
              </div>
            </div>
          </DialogContent>
        </Dialog>
      )}

      {showDialog && !editingWorkflow && (
        <Dialog open={showDialog && !editingWorkflow} onOpenChange={(open) => { setShowDialog(open); if (!open) resetForm(); }}>
          <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
            <DialogHeader><DialogTitle>Create New Workflow</DialogTitle></DialogHeader>
            <form onSubmit={handleSubmit} className="space-y-4">
              <div>
                <Label>Workflow Name *</Label>
                <Input value={formData.workflow_name} onChange={(e) => setFormData({...formData, workflow_name: e.target.value})} placeholder="e.g., New Estimate Follow-up" required />
              </div>
              <div>
                <Label>Description</Label>
                <Textarea value={formData.description} onChange={(e) => setFormData({...formData, description: e.target.value})} rows={3} />
              </div>
              <div>
                <Label>Folder</Label>
                <Input value={formData.folder || ""} onChange={(e) => setFormData({ ...formData, folder: e.target.value })} placeholder="e.g., Nurture, Leads, Invoices" />
              </div>
              <div>
                <Label>Trigger *</Label>
                <Select value={formData.trigger_type} onValueChange={(value) => setFormData({...formData, trigger_type: value})}>
                  <SelectTrigger><SelectValue /></SelectTrigger>
                  <SelectContent>
                    <SelectItem value="estimate_created">Estimate Created</SelectItem>
                    <SelectItem value="estimate_accepted">Estimate Accepted</SelectItem>
                    <SelectItem value="estimate_declined">Estimate Declined</SelectItem>
                    <SelectItem value="estimate_sent">Estimate Sent</SelectItem>
                    <SelectItem value="invoice_created">Invoice Created</SelectItem>
                    <SelectItem value="invoice_paid">Invoice Paid</SelectItem>
                    <SelectItem value="invoice_overdue">Invoice Overdue</SelectItem>
                    <SelectItem value="lead_created">Lead Created</SelectItem>
                    <SelectItem value="lead_status_changed">Lead Status Changed</SelectItem>
                    <SelectItem value="customer_created">Customer Created</SelectItem>
                    <SelectItem value="customer_updated">Customer Updated</SelectItem>
                    <SelectItem value="appointment_created">Appointment Created</SelectItem>
                    <SelectItem value="appointment_completed">Appointment Completed</SelectItem>
                    <SelectItem value="project_started">Project Started</SelectItem>
                    <SelectItem value="project_completed">Project Completed</SelectItem>
                    <SelectItem value="payment_received">Payment Received</SelectItem>
                    <SelectItem value="task_assigned">Task Assigned</SelectItem>
                    <SelectItem value="task_completed">Task Completed</SelectItem>
                    <SelectItem value="proposal_created">Proposal Created</SelectItem>
                    <SelectItem value="proposal_accepted">Proposal Accepted</SelectItem>
                    <SelectItem value="inspection_completed">Inspection Completed</SelectItem>
                    <SelectItem value="ai_communication_logged">AI Communication Logged</SelectItem>
                  </SelectContent>
                </Select>
              </div>
              <div className="flex items-center gap-3">
                <Switch checked={formData.is_active} onCheckedChange={(checked) => setFormData({...formData, is_active: checked})} />
                <Label>Active</Label>
              </div>
              <div className="flex justify-end gap-3 pt-4">
                <Button type="button" variant="outline" onClick={() => { setShowDialog(false); resetForm(); }}>Cancel</Button>
                <Button type="submit" className="bg-blue-600 hover:bg-blue-700">Create Workflow</Button>
              </div>
            </form>
          </DialogContent>
        </Dialog>
      )}

      {runningNow > 0 && (
        <Card className="mb-6">
          <CardHeader>
            <CardTitle className="flex items-center gap-2">
              <Activity className="w-5 h-5 animate-pulse text-green-600" />
              Currently Running ({runningNow})
            </CardTitle>
          </CardHeader>
          <CardContent>
            <div className="space-y-3">
              {executions.slice(0, 5).map((execution) => (
                <div key={execution.id} className="flex items-center justify-between p-4 border rounded-lg bg-green-50">
                  <div>
                    <p className="font-medium text-gray-900">{execution.entity_name}</p>
                    <p className="text-sm text-gray-600">{execution.workflow_name}</p>
                    <p className="text-xs text-gray-500 mt-1">
                      Step {execution.current_step + 1} • Next action: {execution.next_action_time ? format(new Date(execution.next_action_time), 'MMM d, h:mm a') : 'Now'}
                    </p>
                  </div>
                  <Badge className="bg-green-100 text-green-700">
                    <Activity className="w-3 h-3 mr-1 animate-pulse" />
                    Running
                  </Badge>
                </div>
              ))}
            </div>
          </CardContent>
        </Card>
      )}

      <Tabs defaultValue="workflows" className="w-full">
        <TabsList className="grid w-full grid-cols-2 mb-4">
          <TabsTrigger value="workflows">Manage Workflows</TabsTrigger>
          <TabsTrigger value="activity">Activity Log</TabsTrigger>
        </TabsList>

        <TabsContent value="workflows">
          <Card>
            <CardHeader><CardTitle>Your Workflows</CardTitle></CardHeader>
            <CardContent>
              <div className="flex flex-col md:flex-row md:items-center md:justify-between gap-3 mb-4">
                <div className="relative flex-1">
                  <Search className="absolute left-3 top-1/2 -translate-y-1/2 text-gray-400 w-4 h-4" />
                  <Input value={searchQuery} onChange={(e) => setSearchQuery(e.target.value)} placeholder="Search name, description, trigger or tag..." className="pl-9" />
                </div>
                <div className="w-full md:w-56">
                  <Select value={folderFilter} onValueChange={setFolderFilter}>
                    <SelectTrigger><SelectValue placeholder="Filter by folder" /></SelectTrigger>
                    <SelectContent>
                      <SelectItem value="all">All folders</SelectItem>
                      {folders.map((f) => (<SelectItem key={f} value={f}>{f}</SelectItem>))}
                    </SelectContent>
                  </Select>
                </div>
              </div>

              {workflows.length === 0 ? (
                <div className="text-center py-12">
                  <Activity className="w-16 h-16 mx-auto mb-4 text-gray-300" />
                  <p className="text-gray-500 font-medium mb-2">No workflows yet</p>
                  <Button onClick={() => setShowDialog(true)} className="bg-blue-600 hover:bg-blue-700">
                    <Plus className="w-4 h-4 mr-2" />Create First Workflow
                  </Button>
                </div>
              ) : (
                <div className="space-y-4">
                  {filteredWorkflows.map((workflow) => (
                    <Card key={workflow.id} className="border-2">
                      <CardContent className="p-6">
                        <div className="flex items-start justify-between">
                          <div className="flex items-start gap-4 flex-1">
                            <div className="p-3 bg-blue-100 rounded-lg">{getTriggerIcon(workflow.trigger_type)}</div>
                            <div className="flex-1">
                              <div className="flex items-center gap-3 mb-2">
                                <h3 className="text-lg font-semibold text-gray-900">{workflow.workflow_name}</h3>
                                {workflow.is_active ? <Badge className="bg-green-100 text-green-700">Active</Badge> : <Badge variant="outline">Inactive</Badge>}
                                {workflow.folder && <Badge variant="outline" className="bg-yellow-50 text-yellow-700"><Folder className="w-3 h-3 mr-1" />{workflow.folder}</Badge>}
                              </div>
                              {workflow.description && <p className="text-sm text-gray-600 mb-3">{workflow.description}</p>}
                              <div className="flex items-center gap-6 text-sm text-gray-500 mb-3">
                                <div className="flex items-center gap-2"><Zap className="w-4 h-4" /><span>Trigger: {workflow.trigger_type.replace(/_/g, ' ')}</span></div>
                                <div className="flex items-center gap-2"><Activity className="w-4 h-4" /><span>{workflow.execution_count || 0} executions</span></div>
                                {workflow.last_executed && <div className="flex items-center gap-2"><Clock className="w-4 h-4" /><span>Last run: {format(new Date(workflow.last_executed), 'MMM d, h:mm a')}</span></div>}
                              </div>

                              {testResults[workflow.id] && (
                                <div className={`mt-3 p-3 rounded-lg border ${testResults[workflow.id].status === 'success' ? 'bg-green-50 border-green-200' : testResults[workflow.id].status === 'error' ? 'bg-red-50 border-red-200' : 'bg-blue-50 border-blue-200'}`}>
                                  <p className="text-xs font-semibold mb-1">{testResults[workflow.id].message}</p>
                                </div>
                              )}
                            </div>
                          </div>

                          <div className="flex items-center gap-2 flex-wrap">
                            <Button variant="outline" size="sm" onClick={() => handleTest(workflow)} disabled={testMutation.isPending} className="bg-green-50 hover:bg-green-100 text-green-700">
                              <Play className="w-4 h-4 mr-2" />Test
                            </Button>
                            <Button variant="outline" size="sm" onClick={() => handleToggle(workflow)}>
                              {workflow.is_active ? <><Pause className="w-4 h-4 mr-2" />Pause</> : <><Play className="w-4 h-4 mr-2" />Activate</>}
                            </Button>
                            <Button variant="outline" size="sm" onClick={() => { setEditingWorkflow(workflow); setFormData(workflow); setShowDialog(true); }} className="bg-blue-50 hover:bg-blue-100 text-blue-700">
                              <Eye className="w-4 h-4 mr-2" />View
                            </Button>
                            <Button variant="outline" size="sm" onClick={() => { setEditingAIWorkflow(workflow); setAIPrompt(workflow.ai_prompt || ""); }} className="bg-indigo-50 hover:bg-indigo-100 text-indigo-700">
                              <Sparkles className="w-4 h-4 mr-2" />Edit with AI
                            </Button>
                            {workflow.company_id === myCompany?.id ? (
                              <Button variant="outline" size="sm" onClick={() => handleEdit(workflow)}>
                                <Edit className="w-4 h-4 mr-2" />Edit Steps
                              </Button>
                            ) : (
                              <Button variant="outline" size="sm" onClick={() => { setEditingWorkflow(null); setFormData({ ...workflow, workflow_name: `${workflow.workflow_name} (Copy)`, company_id: myCompany?.id }); setShowDialog(true); setShowActionsEditor(true); }} className="bg-purple-50 hover:bg-purple-100 text-purple-700">
                                <Copy className="w-4 h-4 mr-2" />Clone & Edit
                              </Button>
                            )}
                            {!workflow.is_default && workflow.company_id === myCompany?.id && (
                              <Button variant="outline" size="sm" onClick={() => handleDelete(workflow.id)} className="text-red-600 hover:text-red-700">
                                <Trash2 className="w-4 h-4" />
                              </Button>
                            )}
                          </div>
                        </div>
                      </CardContent>
                    </Card>
                  ))}
                </div>
              )}
            </CardContent>
          </Card>
        </TabsContent>

        <TabsContent value="activity">
          <WorkflowActivityLog />
        </TabsContent>
      </Tabs>

      {/* Edit with AI Dialog */}
      <Dialog open={!!editingAIWorkflow} onOpenChange={(open) => !open && setEditingAIWorkflow(null)}>
        <DialogContent className="max-w-xl">
          <DialogHeader>
            <DialogTitle className="flex items-center gap-2 text-xl">
              <Bot className="w-6 h-6 text-indigo-600" />
              Edit with AI: {editingAIWorkflow?.workflow_name}
            </DialogTitle>
          </DialogHeader>
          <div className="space-y-4 py-4">
            <div>
              <Label>Your Request</Label>
              <Textarea value={aiPrompt} onChange={(e) => setAIPrompt(e.target.value)} placeholder="e.g., Change the delay to 10 minutes..." rows={4} className="mt-2 text-base" />
            </div>
            <div className="flex justify-end gap-3 pt-2">
              <Button variant="ghost" onClick={() => setEditingAIWorkflow(null)}>Cancel</Button>
              <Button
                onClick={async () => {
                  if (!aiPrompt.trim()) return;
                  setIsGenerating(true);
                  try {
                    const contextMessage = `Current Workflow Data:\n${JSON.stringify(editingAIWorkflow, null, 2)}\n\nUser Request for this workflow (ID: ${editingAIWorkflow.id}):\n${aiPrompt}`;
                    const response = await base44.functions.invoke('workflowBuilderAgent', { message: contextMessage, user_prompt: aiPrompt, companyId: myCompany?.id });
                    if (response.data.actions_executed?.some(a => a.tool_name === 'update_workflow')) {
                      queryClient.invalidateQueries({ queryKey: ['workflows'] });
                      setEditingAIWorkflow(null);
                      setAIPrompt("");
                      const action = response.data.actions_executed.find(a => a.tool_name === 'update_workflow');
                      alert(action.result || 'Workflow updated successfully!');
                    } else {
                      alert('AI response: ' + response.data.response);
                    }
                  } catch (error) { alert('Error: ' + error.message); }
                  finally { setIsGenerating(false); }
                }}
                disabled={isGenerating || !aiPrompt.trim()}
                className="bg-indigo-600 hover:bg-indigo-700"
              >
                {isGenerating ? <><Activity className="w-4 h-4 mr-2 animate-spin" />Updating...</> : <><Sparkles className="w-4 h-4 mr-2" />Update Workflow</>}
              </Button>
            </div>
          </div>
        </DialogContent>
      </Dialog>

      {/* AI Workflow Builder Dialog */}
      <Dialog open={showAIPrompt} onOpenChange={setShowAIPrompt}>
        <DialogContent className="max-w-xl">
          <DialogHeader>
            <DialogTitle className="flex items-center gap-2 text-xl">
              <Bot className="w-6 h-6 text-indigo-600" />
              AI Workflow Builder
            </DialogTitle>
          </DialogHeader>
          <div className="space-y-4 py-4">
            <div>
              <Label>Describe your workflow</Label>
              <Textarea value={aiPrompt} onChange={(e) => setAIPrompt(e.target.value)} placeholder="e.g., When a new customer is created, send them a welcome email..." rows={4} className="mt-2 text-base" />
            </div>
            <div className="flex justify-end gap-3 pt-2">
              <Button variant="ghost" onClick={() => setShowAIPrompt(false)}>Cancel</Button>
              <Button
                onClick={async () => {
                  if (!aiPrompt.trim()) return;
                  setIsGenerating(true);
                  try {
                    const response = await base44.functions.invoke('workflowBuilderAgent', { message: aiPrompt, user_prompt: aiPrompt, companyId: myCompany?.id });
                    if (response.data.actions_executed?.some(a => a.tool_name === 'create_workflow')) {
                      queryClient.invalidateQueries({ queryKey: ['workflows'] });
                      setShowAIPrompt(false);
                      setAIPrompt("");
                      const action = response.data.actions_executed.find(a => a.tool_name === 'create_workflow');
                      alert(action.result || 'Workflow created successfully!');
                    } else {
                      alert('Lexi response: ' + response.data.response);
                    }
                  } catch (error) { alert('Error: ' + error.message); }
                  finally { setIsGenerating(false); }
                }}
                disabled={isGenerating || !aiPrompt.trim()}
                className="bg-indigo-600 hover:bg-indigo-700"
              >
                {isGenerating ? <><Activity className="w-4 h-4 mr-2 animate-spin" />Building...</> : <><Sparkles className="w-4 h-4 mr-2" />Generate Workflow</>}
              </Button>
            </div>
          </div>
        </DialogContent>
      </Dialog>

      {/* Actions Editor Dialog */}
      <Dialog open={showActionsEditor} onOpenChange={(open) => { setShowActionsEditor(open); if (!open) { setShowDialog(false); resetForm(); } }}>
        <DialogContent className="max-w-4xl max-h-[90vh] overflow-y-auto">
          <DialogHeader>
            <DialogTitle className="flex items-center gap-2">
              <Settings className="w-5 h-5" />
              Edit Workflow Steps: {formData.workflow_name}
            </DialogTitle>
          </DialogHeader>
          <form onSubmit={handleSubmit} className="space-y-6">
            <div className="grid grid-cols-2 gap-4">
              <div>
                <Label>Workflow Name *</Label>
                <Input value={formData.workflow_name} onChange={(e) => setFormData({...formData, workflow_name: e.target.value})} required />
              </div>
              <div>
                <Label>Trigger *</Label>
                <Select value={formData.trigger_type} onValueChange={(value) => setFormData({...formData, trigger_type: value})}>
                  <SelectTrigger><SelectValue /></SelectTrigger>
                  <SelectContent>
                    <SelectItem value="estimate_created">Estimate Created</SelectItem>
                    <SelectItem value="estimate_accepted">Estimate Accepted</SelectItem>
                    <SelectItem value="invoice_created">Invoice Created</SelectItem>
                    <SelectItem value="lead_created">Lead Created</SelectItem>
                    <SelectItem value="customer_created">Customer Created</SelectItem>
                    <SelectItem value="payment_received">Payment Received</SelectItem>
                    <SelectItem value="project_started">Project Started</SelectItem>
                  </SelectContent>
                </Select>
              </div>
            </div>
            <div>
              <Label>Description</Label>
              <Textarea value={formData.description} onChange={(e) => setFormData({...formData, description: e.target.value})} rows={2} />
            </div>
            <div>
              <Label>Folder (optional)</Label>
              <Input value={formData.folder || ""} onChange={(e) => setFormData({ ...formData, folder: e.target.value })} placeholder="Group this workflow into a folder" />
            </div>

            <div className="border-t pt-4">
              <div className="flex items-center justify-between mb-4">
                <h3 className="text-lg font-semibold">Automation Steps</h3>
                <Button type="button" onClick={addAction} size="sm" variant="outline">
                  <Plus className="w-4 h-4 mr-2" />Add Step
                </Button>
              </div>

              {formData.actions && formData.actions.length > 0 ? (
                <div className="space-y-4">
                  {formData.actions.map((action, index) => (
                    <Card key={index} className="bg-gray-50">
                      <CardContent className="p-4 space-y-3">
                        <div className="flex items-center justify-between mb-3">
                          <Badge>Step {index + 1}</Badge>
                          <Button type="button" variant="ghost" size="sm" onClick={() => removeAction(index)} className="text-red-600">
                            <Trash2 className="w-4 h-4" />
                          </Button>
                        </div>
                        <div className="grid grid-cols-2 gap-3">
                          <div>
                            <Label className="text-xs">Action Type</Label>
                            <Select value={action.action_type || ''} onValueChange={(value) => updateAction(index, 'action_type', value)}>
                              <SelectTrigger><SelectValue /></SelectTrigger>
                              <SelectContent>
                                <SelectItem value="send_email">Send Email</SelectItem>
                                <SelectItem value="send_sms">Send SMS</SelectItem>
                                <SelectItem value="create_task">Create Task</SelectItem>
                                <SelectItem value="wait">Wait</SelectItem>
                              </SelectContent>
                            </Select>
                          </div>
                          <div>
                            <Label className="text-xs">Delay (minutes)</Label>
                            <Input type="number" value={action.delay_minutes || 0} onChange={(e) => updateAction(index, 'delay_minutes', parseInt(e.target.value))} />
                          </div>
                        </div>
                        {action.action_type === 'send_email' && (
                          <>
                            <div>
                              <Label className="text-xs">Recipient Email(s) *</Label>
                              <Input type="text" value={action.recipient || ''} onChange={(e) => updateAction(index, 'recipient', e.target.value)} placeholder="email@example.com or 'lead', 'customer'" />
                            </div>
                            <div>
                              <Label className="text-xs">Subject</Label>
                              <Input value={action.email_subject || ''} onChange={(e) => updateAction(index, 'email_subject', e.target.value)} />
                            </div>
                            <div>
                              <Label className="text-xs">Email Body</Label>
                              <Textarea value={action.email_body || ''} onChange={(e) => updateAction(index, 'email_body', e.target.value)} rows={4} />
                            </div>
                          </>
                        )}
                        {action.action_type === 'send_sms' && (
                          <>
                            <div>
                              <Label className="text-xs">SMS Message</Label>
                              <Textarea value={action.sms_message || ''} onChange={(e) => updateAction(index, 'sms_message', e.target.value)} rows={3} />
                            </div>
                            <div>
                              <Label className="text-xs">Recipient</Label>
                              <Input value={action.recipient || 'lead'} onChange={(e) => updateAction(index, 'recipient', e.target.value)} />
                            </div>
                          </>
                        )}
                        {action.action_type === 'create_task' && (
                          <>
                            <div>
                              <Label className="text-xs">Task Title</Label>
                              <Input value={action.task_title || ''} onChange={(e) => updateAction(index, 'task_title', e.target.value)} />
                            </div>
                            <div>
                              <Label className="text-xs">Task Description</Label>
                              <Textarea value={action.task_description || ''} onChange={(e) => updateAction(index, 'task_description', e.target.value)} rows={3} />
                            </div>
                          </>
                        )}
                      </CardContent>
                    </Card>
                  ))}
                </div>
              ) : (
                <div className="text-center py-8 bg-gray-50 rounded-lg border-2 border-dashed">
                  <p className="text-gray-500 mb-3">No automation steps yet</p>
                  <Button type="button" onClick={addAction} variant="outline"><Plus className="w-4 h-4 mr-2" />Add First Step</Button>
                </div>
              )}
            </div>

            <div className="flex items-center gap-3 pt-4 border-t">
              <Switch checked={formData.is_active} onCheckedChange={(checked) => setFormData({...formData, is_active: checked})} />
              <Label>Active</Label>
            </div>
            <div className="flex justify-end gap-3 pt-4">
              <Button type="button" variant="outline" onClick={() => { setShowActionsEditor(false); setShowDialog(false); resetForm(); }}>Cancel</Button>
              <Button type="submit" className="bg-blue-600 hover:bg-blue-700"><Save className="w-4 h-4 mr-2" />Save Workflow</Button>
            </div>
          </form>
        </DialogContent>
      </Dialog>
    </div>
  );
}
